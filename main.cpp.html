#include <iostream>

void loadDataFromFile(Customer customer[], Account account[]);

        cout << "Press [3] for Search an account" << endl;
        cout << "Press [4] exit " << endl;

        switch (option) {
            case 1:
            {

                    displayAccountSearch(customer, account);

{
    AccID = id;
    Saving_Current = s_c;
    creation_date = cd;

    CNIC = cnic;

int Account::search(string id_input)
{
    for (int i = 0; i < numclients; i++)
    {
        if (id_input == account[i].getAccID())
        {
            search_position = i;
            return search_position;
        }
    }
    search_position = -1;
    return search_position;
}

{
    return balance;
}

{
    balance -= amount;
}

void loadDataFromFile(Customer customer[], Account account[])
{
    ifstream read("lol.txt");

    if (!read.is_open())
    {
        cout << "Error: File could not be opened." << endl;
        return;
    }

    cout << setw(10) << "First Name" << setw(20) << "Last Name" << setw(20) << "CNIC" << setw(20) << "ACCID" << setw(20) << "Saving/current" << setw(20) << "Balance" << setw(20) << "Creation date" << endl;
    cout << "---------------------------------------------------------------------------------------------------------------------------------------------------" << endl;

    string fname, lname, cn, id, s_c, cd;
    int bal;

    int i = 0;
    string format;
    getline(read, format);

    while (i < max_clients && read >> fname >> lname >> cn >> id >> s_c >> bal >> cd)
    {
        customer[i].input_values(fname, lname, cn);
        account[i].setupA(id, s_c, cd, bal, cn);
        cout << setw(10) << fname << setw(20) << lname << setw(20) << cn << setw(20) << id << setw(20) << s_c << setw(20) << bal << setw(20) << cd << endl;
        i++;
    }

    numclients = i;

    if (i == 0)
    {
        cout << "Error: No data found in the file." << endl;
    }

    read.close();
}

{

void transaction_amount(Account &account, int withdraw)
{
    if (account.getBalance() >= withdraw)
    {
        account.updateBalance(withdraw);
        cout << "Your remaining balance after withdrawing " << withdraw << " is " << account.getBalance() << endl;

        Transaction transaction;
        transaction.createTransaction("123456", "Debit", withdraw, "18/03/2024", account.getAccID());

        transaction.saveTransactionToFile(customer[search_position], account, transaction);
    } else
    {
        cout << "Insufficient balance." << endl;
    }
}

void Transaction::createTransaction(string trID, string debitCredit, int amt, string dateTime, string accID)
{
    TrID = trID;
    debit_credit = debitCredit;
    amount = amt;
    date_time = dateTime;
    AccID = accID;
}

void Transaction::saveTransactionToFile(Customer customer, Account account, Transaction transaction)
{
    ofstream write("transaction.txt", ios::app);

    if (!write.is_open())
    {
        cout << "Error: File could not be opened." << endl;
        return;
    }

    write << "Transaction ID: " << transaction.TrID << endl;
    write << "Customer Name: " << customer.returnfname() << " " << customer.returnlname() << endl;
    write << "Customer CNIC: " << customer.getCNIC() << endl;
    write << "Account ID: " << account.getAccID() << endl;
    write << "Transaction Type: " << transaction.debit_credit << endl;
    write << "Transaction Amount: " << transaction.amount << endl;
    write << "Date Time: " << transaction.date_time << endl;

    write.close();
}
